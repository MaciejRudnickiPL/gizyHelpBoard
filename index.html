// DEFINICJA PANELU
        var board = new HelpBoard('myBoard');

        // DEFINICJA MODUŁÓW
        var info = new HelpInfo('Info');                 // definicja modułu wyświetlającego zmienne
        var field = new HelpFields('Pole');              // definicja modułu wyświetlającego zmienne, obiekty i tablice
        var stoper = new HelpStoper('Stoper');           // definicja modułu wyświetlającego stoper
        var list = new HelpInfoList('Lista');            // definicja modułu wyświetlającego numerowaną listę zmiennych
        var btn = new HelpButton('Test Btn', testBtn);   // definicja przycisku uruchamiającego funkcje

        // DODAWANIE MODUŁÓW DO PANELU
        board.add(info);
        board.add(field);
        board.add(stoper);
        board.add(list);
        board.add(btn)
